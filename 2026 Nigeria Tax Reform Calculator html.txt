<script>
/* ---------- TAB SWITCH ---------- */
document.querySelectorAll(".tab").forEach(tab => {
    tab.addEventListener("click", () => {
        document.querySelectorAll(".tab").forEach(t => t.classList.remove("active"));
        tab.classList.add("active");

        document.querySelectorAll(".section").forEach(s => s.classList.remove("active"));
        document.getElementById(tab.dataset.tab).classList.add("active");

        calculate();
    });
});

/* ---------- CALCULATION LOGIC ---------- */
function calcTax(annualTaxable) {
    let t = annualTaxable;
    let tax = 0;

    const bands = [
        {limit:800000, rate:0},
        {limit:2200000, rate:0.15},
        {limit:9000000, rate:0.18},
        {limit:13000000, rate:0.21},
        {limit:25000000, rate:0.23},
        {limit:Infinity, rate:0.25}
    ];

    for (const b of bands) {
        if (t <= 0) break;
        let use = Math.min(t, b.limit);
        tax += use * b.rate;
        t -= use;
    }
    return tax;
}

function calculate() {
    const mode = document.getElementById("modeToggle").value;
    let gross = 0, reliefs = 0, taxable = 0;

    /* ----- Salary Earner ----- */
    if (document.getElementById("salary").classList.contains("active")) {
        let g = Number(document.getElementById("salaryGross").value) || 0;
        let rent = Number(document.getElementById("salaryRent").value) || 0;

        gross = g;

        let pension = document.getElementById("chkPension").checked ? g * 0.08 : 0;
        let nhf = document.getElementById("chkNHF").checked ? g * 0.025 : 0;
        let nhis = document.getElementById("chkNHIS").checked ? g * 0.05 : 0;

        let rentRelief = Math.min(rent * 0.20, 500000);

        reliefs = pension + nhf + nhis + rentRelief;
        taxable = g - reliefs;
    }

    /* ----- Business Owner ----- */
    else {
        let t = Number(document.getElementById("bizTurnover").value) || 0;
        let e = Number(document.getElementById("bizExpenses").value) || 0;
        gross = t;
        taxable = t - e;
        reliefs = e; // For display
    }

    if (taxable < 0) taxable = 0;

    let annualTax = calcTax(taxable);
    let takeHome = gross - annualTax;

    if (mode === "monthly") {
        gross /= 12;
        reliefs /= 12;
        taxable /= 12;
        annualTax /= 12;
        takeHome /= 12;
    }

    document.getElementById("rGross").innerHTML = "₦" + gross.toLocaleString();
    document.getElementById("rReliefs").innerHTML = "₦" + reliefs.toLocaleString();
    document.getElementById("rTaxable").innerHTML = "₦" + taxable.toLocaleString();
    document.getElementById("rTax").innerHTML = "₦" + annualTax.toLocaleString();
    document.getElementById("rTakehome").innerHTML = "₦" + takeHome.toLocaleString();
}

/* ---------- LIVE UPDATES ---------- */
document.querySelectorAll("input, select").forEach(el => {
    el.addEventListener("input", calculate);
});

calculate();
